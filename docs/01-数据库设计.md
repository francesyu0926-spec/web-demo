# 观点科技电子招投标交易平台 — 数据库设计

> 技术栈：Java + Spring Boot + MySQL 8.0（InnoDB / utf8mb4）
> 命名规范：表名小写下划线，主键统一 `id`（BIGINT，雪花/自增），所有业务表带 `create_time / update_time / deleted` 逻辑删除字段。

---

## 0. 通用约定

- 所有表默认包含审计字段：

| 字段 | 类型 | 说明 |
|---|---|---|
| `id` | BIGINT UNSIGNED PK | 主键 |
| `create_by` | BIGINT | 创建人用户ID |
| `create_time` | DATETIME | 创建时间 |
| `update_by` | BIGINT | 更新人用户ID |
| `update_time` | DATETIME | 更新时间 |
| `deleted` | TINYINT(1) | 逻辑删除 0/1 |

- 金额统一用 `DECIMAL(15,2)`；状态/类型字段用 `TINYINT` + 字典或 `VARCHAR` 枚举码（下文每张表标注取值）。
- 文件不直接入库，统一走 `attachment` 表存元数据 + 对象存储（MinIO/OSS）。

## 1. 用户与权限

### sys_user 用户
```sql
CREATE TABLE sys_user (
  id           BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  username     VARCHAR(64)  NOT NULL COMMENT '登录账号',
  password     VARCHAR(128) NULL COMMENT 'BCrypt密码,微信注册可空',
  real_name    VARCHAR(64)  NULL COMMENT '姓名',
  phone        VARCHAR(20)  NULL COMMENT '手机号',
  email        VARCHAR(128) NULL,
  wx_openid    VARCHAR(64)  NULL COMMENT '微信openid',
  wx_unionid   VARCHAR(64)  NULL,
  avatar       VARCHAR(255) NULL,
  status       TINYINT      NOT NULL DEFAULT 1 COMMENT '0禁用 1正常',
  create_time  DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  UNIQUE KEY uk_username(username),
  UNIQUE KEY uk_phone(phone),
  KEY idx_openid(wx_openid)
) COMMENT='用户';
```

### sys_role 角色 / sys_user_role 用户-角色（多对多，支持角色切换）
```sql
CREATE TABLE sys_role (
  id    BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  code  VARCHAR(32) NOT NULL COMMENT 'GUEST/BIDDER/AGENT/EXPERT/ADMIN',
  name  VARCHAR(32) NOT NULL,
  UNIQUE KEY uk_code(code)
) COMMENT='角色';

CREATE TABLE sys_user_role (
  id       BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id  BIGINT NOT NULL,
  role_id  BIGINT NOT NULL,
  audit_status TINYINT DEFAULT 1 COMMENT '0待审核 1已通过(专家角色需审核)',
  UNIQUE KEY uk_user_role(user_id, role_id)
) COMMENT='用户角色关系';
```
> 角色码：`GUEST 游客` / `BIDDER 投标人` / `AGENT 项目经理(招标代理)` / `EXPERT 评标专家` / `ADMIN 平台管理员`。

### supplier_profile 投标人(供应商)资料
```sql
CREATE TABLE supplier_profile (
  id            BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id       BIGINT NOT NULL,
  company_name  VARCHAR(128) NOT NULL COMMENT '企业名称',
  credit_code   VARCHAR(32)  NOT NULL COMMENT '统一社会信用代码',
  legal_person  VARCHAR(64),
  contact_name  VARCHAR(64),
  contact_phone VARCHAR(20),
  license_file_id BIGINT COMMENT '营业执照附件',
  status        TINYINT DEFAULT 0 COMMENT '0待认证 1已认证 2驳回',
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  UNIQUE KEY uk_credit(credit_code)
) COMMENT='投标人企业资料';
```

### expert_profile 专家资料 / expert_application 专家申请
```sql
CREATE TABLE expert_profile (
  id          BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id     BIGINT NOT NULL,
  expert_no   VARCHAR(32) COMMENT '专家编号',
  major       VARCHAR(128) COMMENT '专业领域(可多个,逗号或子表)',
  title       VARCHAR(64)  COMMENT '职称',
  id_card     VARCHAR(32),
  sign_img_id BIGINT COMMENT '电子签名图(暂缓)',
  status      TINYINT DEFAULT 1 COMMENT '0停用 1正常',
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  UNIQUE KEY uk_user(user_id)
) COMMENT='专家资料';

CREATE TABLE expert_application (
  id          BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id     BIGINT NOT NULL,
  major       VARCHAR(128),
  attach_id   BIGINT COMMENT '证明材料',
  audit_status TINYINT DEFAULT 0 COMMENT '0待审 1通过 2驳回',
  audit_remark VARCHAR(255),
  audit_by    BIGINT, audit_time DATETIME,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0
) COMMENT='专家入驻申请';
```

## 2. 招标侧

### tender_project 招标项目（核心表 / 带状态机）
```sql
CREATE TABLE tender_project (
  id            BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  project_no    VARCHAR(40)  NOT NULL COMMENT '项目编号',
  name          VARCHAR(200) NOT NULL COMMENT '项目名称',
  category      VARCHAR(32)  COMMENT '招标方式:公开/邀请/竞争性谈判...',
  industry      VARCHAR(32)  COMMENT '行业分类',
  region        VARCHAR(64)  COMMENT '地区',
  budget        DECIMAL(15,2) COMMENT '预算/控制价',
  deposit_amount DECIMAL(15,2) COMMENT '保证金金额',
  agent_id      BIGINT NOT NULL COMMENT '招标代理(项目经理)user_id',
  status        VARCHAR(20) NOT NULL DEFAULT 'DRAFT' COMMENT '见状态机',
  reg_start     DATETIME COMMENT '报名开始',
  reg_end       DATETIME COMMENT '报名截止',
  bid_open_time DATETIME COMMENT '开标时间',
  content       MEDIUMTEXT COMMENT '招标公告正文',
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  UNIQUE KEY uk_no(project_no),
  KEY idx_status(status), KEY idx_agent(agent_id)
) COMMENT='招标项目';
```
**status 状态机**：`DRAFT 草稿` → `PUBLISHED 报名中` → `REG_CLOSED 报名截止` → `OPENING/OPENED 开标` → `EVALUATING 评标中` → `WIN_PUBLICITY 中标公示` → `FINISHED 结束`；任意阶段可 `ABORTED 废标/流标`。

### announcement 公告
```sql
CREATE TABLE announcement (
  id          BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  project_id  BIGINT NOT NULL,
  type        TINYINT NOT NULL COMMENT '1招标公告 2中标候选人公示 3中标结果',
  title       VARCHAR(200),
  content     MEDIUMTEXT,
  publish_time DATETIME,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  KEY idx_project_type(project_id, type)
) COMMENT='公告(招标/中标候选人/中标结果)';
```

## 3. 投标侧

### bid_registration 投标报名 / bid_document 投标文件 / deposit 保证金
```sql
CREATE TABLE bid_registration (
  id            BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  project_id    BIGINT NOT NULL,
  supplier_id   BIGINT NOT NULL COMMENT '投标人user_id/供应商',
  audit_status  TINYINT DEFAULT 0 COMMENT '0待审核 1通过 2驳回',
  audit_remark  VARCHAR(255),
  bid_status    VARCHAR(20) DEFAULT 'REGISTERED'
                COMMENT 'REGISTERED报名/SUBMITTED已投/WON中标/LOST未中/INVALID废标',
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  UNIQUE KEY uk_project_supplier(project_id, supplier_id)
) COMMENT='投标报名';

CREATE TABLE bid_document (
  id              BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  registration_id BIGINT NOT NULL,
  attach_id       BIGINT NOT NULL COMMENT '投标文件附件',
  bid_price       DECIMAL(15,2) COMMENT '投标报价',
  submit_time     DATETIME,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  KEY idx_reg(registration_id)
) COMMENT='投标文件(开标前不可下载)';

CREATE TABLE deposit (
  id              BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  registration_id BIGINT NOT NULL,
  amount          DECIMAL(15,2),
  status          TINYINT DEFAULT 0 COMMENT '0未缴 1已缴 2已退',
  pay_time DATETIME, refund_time DATETIME,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0
) COMMENT='保证金(原型标注待定,先预留)';
```

## 4. 开评标

### expert_assignment 专家抽取/指派
```sql
CREATE TABLE expert_assignment (
  id          BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  project_id  BIGINT NOT NULL,
  expert_id   BIGINT NOT NULL COMMENT '专家user_id',
  status      VARCHAR(20) DEFAULT 'PENDING'
              COMMENT 'PENDING待确认/ACCEPTED已接受/REJECTED已拒绝/SIGNED已签到',
  respond_time DATETIME, sign_time DATETIME,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  UNIQUE KEY uk_project_expert(project_id, expert_id)
) COMMENT='评标专家抽取';
```

### evaluation_item 评审项 / evaluation_score 评分
```sql
CREATE TABLE evaluation_item (
  id          BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  project_id  BIGINT NOT NULL,
  type        VARCHAR(20) NOT NULL
              COMMENT 'FORMAL形式/QUALIFY资格/RESPONSE响应性/COMMERCE商务/TECH技术',
  name        VARCHAR(128),
  max_score   DECIMAL(6,2),
  weight      DECIMAL(5,2) COMMENT '权重%',
  sort        INT,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  KEY idx_project(project_id)
) COMMENT='评审项配置(五类评审)';

CREATE TABLE evaluation_score (
  id              BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  project_id      BIGINT NOT NULL,
  registration_id BIGINT NOT NULL COMMENT '被评投标人',
  expert_id       BIGINT NOT NULL COMMENT '评分专家',
  item_id         BIGINT NOT NULL COMMENT '评审项',
  pass            TINYINT COMMENT '形式/资格/响应性: 0否1是',
  score           DECIMAL(6,2) COMMENT '商务/技术评分',
  remark          VARCHAR(255),
  submitted       TINYINT DEFAULT 0 COMMENT '0暂存 1已提交',
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  UNIQUE KEY uk_score(registration_id, expert_id, item_id)
) COMMENT='评分记录(专家×投标人×评审项)';
```

### award 中标/定标
```sql
CREATE TABLE award (
  id              BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  project_id      BIGINT NOT NULL,
  registration_id BIGINT NOT NULL COMMENT '中标投标人',
  rank            INT COMMENT '中标候选人排名(1=第一)',
  final_score     DECIMAL(8,2) COMMENT '加权最终得分',
  is_winner       TINYINT DEFAULT 0 COMMENT '是否最终中标',
  publicity_start DATETIME, publicity_end DATETIME COMMENT '公示期',
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  KEY idx_project(project_id)
) COMMENT='中标候选人/定标';
```

## 5. 通用

```sql
CREATE TABLE attachment (
  id          BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  file_name   VARCHAR(255),
  file_key    VARCHAR(255) COMMENT '对象存储key',
  file_size   BIGINT,
  content_type VARCHAR(128),
  biz_type    VARCHAR(32) COMMENT '业务类型:bid_doc/license/article...',
  create_by BIGINT, create_time DATETIME, deleted TINYINT DEFAULT 0
) COMMENT='附件元数据';

CREATE TABLE notification (
  id        BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id   BIGINT NOT NULL COMMENT '接收人',
  type      VARCHAR(32) COMMENT 'AUDIT/INVITE/OPEN/AWARD...',
  title     VARCHAR(128),
  content   VARCHAR(500),
  biz_id    BIGINT COMMENT '关联业务id',
  is_read   TINYINT DEFAULT 0,
  create_time DATETIME,
  KEY idx_user_read(user_id, is_read)
) COMMENT='消息通知';

CREATE TABLE complaint (
  id        BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id   BIGINT, project_id BIGINT NULL,
  type      VARCHAR(32) COMMENT '投诉/建议',
  content   TEXT, attach_id BIGINT,
  status    TINYINT DEFAULT 0 COMMENT '0待处理 1处理中 2已回复',
  reply     TEXT,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0
) COMMENT='投诉建议';

CREATE TABLE article_category (
  id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  code VARCHAR(32) COMMENT 'guide业务指南/policy政策法规/download下载专区',
  name VARCHAR(64)
) COMMENT='内容栏目';

CREATE TABLE article (
  id          BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  category_id BIGINT NOT NULL,
  title       VARCHAR(200),
  content     MEDIUMTEXT,
  attach_id   BIGINT COMMENT '下载专区文件',
  publish_time DATETIME,
  create_time DATETIME, update_time DATETIME, deleted TINYINT DEFAULT 0,
  KEY idx_category(category_id)
) COMMENT='业务指南/政策法规/下载专区';

CREATE TABLE operation_log (
  id        BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id   BIGINT, role_code VARCHAR(32),
  module    VARCHAR(64), action VARCHAR(64),
  biz_id    BIGINT, detail VARCHAR(1000), ip VARCHAR(64),
  create_time DATETIME,
  KEY idx_biz(module, biz_id)
) COMMENT='操作审计日志(合规必备)';
```

## 6. 表关系速览（ER 概要）

```text
sys_user 1—N sys_user_role N—1 sys_role
sys_user 1—1 supplier_profile / expert_profile
tender_project 1—N announcement
tender_project 1—N bid_registration 1—N bid_document
bid_registration 1—1 deposit
tender_project 1—N expert_assignment (N—1 sys_user[expert])
tender_project 1—N evaluation_item
evaluation_score: registration × expert × item (三维唯一)
tender_project 1—N award 1—1 bid_registration
```
